(2)产生100个随机数并使用冒泡排序和选择排序对这些数进行排序。

package HomeWork1;

import java.util.Random;

public class ArraysAndrandom {
    static int [] a = new int[110];
    static int [] b = new int[110];

    static void bubbleSort(){
        for(int i = 1 ; i < 100; i ++){
            for(int j = 1 ; j <= 100 - i; j ++){
                if(a[j] > a[j + 1])
                {
                    int t = a[j];
                    a[j] = a[j + 1];
                    a[j + 1] = t;
                }
            }
        }
    }

    static void selectS(){
        for(int i = 1 ; i <= 100; i ++){
            for(int j = i + 1; j <= 100; j++){
                if(b[i] > b[j]){
                    int t = b[i];
                    b[i] = b[j];
                    b[j] = t;
                }
            }
        }
    }

    public static void main(String[] args) {
        Random random = new Random();

        for(int i = 1 ;i <= 100; i ++) {
            a[i] = random.nextInt(1000) + 1;
            b[i] = random.nextInt(1000) + 1;
        }
        bubbleSort();
        selectS();
        System.out.println("冒泡排序后的结果：");
        for(int i = 1 ;i <= 100; i ++) System.out.printf("%d ",a[i]);
        System.out.println();

        System.out.println("选择排序后的结果：");
        for(int i = 1 ; i <= 100; i ++) System.out.printf("%d ",b[i]);
        System.out.println();
    }

}

(3)利用二分查找算法对一个有序数列进行查找

package HomeWork1;

import java.util.Arrays;
import java.util.Scanner;

public class midfind {
    static int N = 100010;
    static int[] a = new int[N];

    public static void main(String[] args) {
        Scanner cin = new Scanner(System.in);
        int n = cin.nextInt();

        for (int i = 0; i < n; i++) {
            a[i] = cin.nextInt();
        }
        Arrays.sort(a,0,n);
        System.out.println("请输入要查找的数");
        int x = cin.nextInt();

        int l = 0, r = n;
        while(l < r){
            int mid = l + r >> 1;
            if(a[mid] >= x) r = mid;
            else l = mid + 1;
        }
        System.out.println("要查找的数在数组的第" + (r + 1)  + "个位置");
    }
}

package HomeWork1;

import java.util.Scanner;

public class 进制转换 {
    static int N = 100010;
    static int[] a = new int[N];
    static int[] b = new int[N];
    static char[] c = new char[N];
    static int[] d = new int[N];
    public static void main(String[] args) {
        Scanner cin = new Scanner(System.in);

        int n = cin.nextInt();
        int temp = n;
        int cnt1 = 0, cnt2 = 0,cnt3 = 0;

        for (int i = 0; temp > 0; i++) {
            a[i] = temp % 2;
            temp /= 2;
            cnt1 ++;
        }
        cnt1 --;
        temp = n;
        for(int j = 0;temp > 0 ;j ++){
            b[j] = n % 8;
            temp /= 8;
            cnt2 ++;
        }
        cnt2 --;
        System.out.println("n的十进制表示为：" + n);
        System.out.printf("n的二进制表示为：");
        for(int i = cnt1 ;i >= 0; i --) System.out.printf("%d",a[i]);
        System.out.println();

        System.out.printf("n的八进制表示为：");
        for (int j = cnt2 ; j >= 0; j --) System.out.printf("%d",b[j]);
        System.out.println();

        temp = n;
        System.out.printf("n的十六进制表示为：");

        for(int i = 0 ; temp != 0; i ++){
            int t = temp % 16;
            if(t > 9) c[i] = (char)(t + 'A' - 10);
            else  c[i] = (char)(t + '0');
            temp /= 16;
            cnt3 ++;
        }
        cnt3 --;
        for(int i = cnt3 ; i >= 0; i --) System.out.printf("%c",c[i]);
        System.out.println();
    }
}

    }

    public static class Pillar{

        double height;

        public Pillar() {
        }

        public Pillar(Geometry bottom, double height) {
            this.bottom = bottom;
            this.height = height;
        }
        public double getV(){
            return bottom.geyArea() * this.height;
        }
    }
    public static class Cricle implements Geometry{
        double radius;

        public Cricle(double radius) {
            this.radius = radius;
        }

        public Cricle() {

        }

        @Override
        public double geyArea() {
            return Math.PI*this.radius * this.radius;
        }
    }
    public static class Rect implements Geometry{
        double wide,length;

        public Rect() {
        }

        public Rect(double wide, double length) {
            this.wide = wide;
            this.length = length;
        }

        @Override
        public double geyArea() {

        }

class PoliceListen implements ActionListener{
    WindowNumber view;

    public void setView(WindowNumber view) {
        this.view = view;
    }
    public void actionPerformed(ActionEvent e){
        int n;
        BigInteger m = BigInteger.ONE;
        BigInteger res = BigInteger.ZERO;
        String str = view.textInput.getText();
        if(!str.isEmpty()){
            try {
                n = Integer.parseInt(str);
                for(int i = 1 ; i <= n; i ++){
                    m = m.multiply(BigInteger.valueOf(i));
                    res = res.add(m);
                }
                view.textShow.setText(" "+res);
            }
            catch (Exception exp){
                view.textInput.setText("请输入数字");
            }
        }
    }
}

package HomeWork5;

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.awt.event.ItemEvent;
import java.awt.event.ItemListener;

class WindowOperation extends JFrame {
    JTextField inputNumberone,inputNumbertwo;
    JComboBox<String> choiceFuhao;
    JTextArea textShow;
    JButton button;
    OperatorListener operator;
    ComputerListener computer;
    public WindowOperation(){
        init();
        setVisible(true);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }
    void init(){
        setLayout(new FlowLayout());
        inputNumberone = new JTextField(5);
        inputNumbertwo = new JTextField(5);
        choiceFuhao = new JComboBox<String>();
        button = new JButton("计算");
        choiceFuhao.addItem("请选择运算符号：");
        String []a = {"+","-","*","/"};
        for (int i = 0; i < a.length; i++) {
            choiceFuhao.addItem(a[i]);
        }
        textShow = new JTextArea(9,30);
        operator = new OperatorListener();
        computer = new ComputerListener();
        operator.setChoice(choiceFuhao);
        operator.setWorkTogether(computer);
        computer.setInpuNumberOne(inputNumberone);
        computer.setInputNumbertwo(inputNumbertwo);
        computer.setTextShow(textShow);
        choiceFuhao.addItemListener(operator);
        button.addActionListener(computer);
        add(inputNumberone);
        add(choiceFuhao);
        add(inputNumbertwo);
        add(button);
        add(new JScrollPane(textShow));
    }

}

class OperatorListener implements ItemListener{
    JComboBox<String> choice;
    ComputerListener workTogether;

    public void setChoice(JComboBox<String> choice) {
        this.choice = choice;
    }

    public void setWorkTogether(ComputerListener workTogether) {
        this.workTogether = workTogether;
    }
    public void itemStateChanged(ItemEvent e){
        String fuhao = choice.getSelectedItem().toString();
        workTogether.setFuhao(fuhao);
    }
}

class ComputerListener implements ActionListener{
    JTextField inpuNumberOne,inputNumbertwo;
    JTextArea textShow;
    String fuhao;

    public void setInpuNumberOne(JTextField inpuNumberOne) {
        this.inpuNumberOne = inpuNumberOne;
    }

    public void setInputNumbertwo(JTextField inputNumbertwo) {
        this.inputNumbertwo = inputNumbertwo;
    }

    public void setTextShow(JTextArea textShow) {
        this.textShow = textShow;
    }

    public void setFuhao(String fuhao) {
        this.fuhao = fuhao;
    }
    public void actionPerformed(ActionEvent e){
        try {
            double number1 = Double.parseDouble(inpuNumberOne.getText());
            double number2 = Double.parseDouble(inputNumbertwo.getText());
            double result = 0;
            if(fuhao.equals("+")){
                result =number1 + number2;
            }
            else if(fuhao.equals("-")){
                result = number1 - number2;
            }
            else if(fuhao.equals("*")){
                result = number1 * number2;
            }
            else if(fuhao.equals("/")){
                result = number1 / number2;
            }
            textShow.append(number1 + "" + fuhao  +"" + number2 + "=" +result + "\n");
        }
        catch (Exception exp){
            textShow.append("\n请输入数字字符\n");
        }
    }
}

class FileAccept implements FilenameFilter{
    private String extendName;
    public void setExtendName(String s){
        extendName = "." + s;
    }
    public boolean accept(File dir,String name){
        return name.endsWith(extendName);

}
