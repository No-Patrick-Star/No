1.基本概念

1.1 流网络，不考虑反向边

在一张有向图中 G = (V, E) 中有一个源点 S 和一个汇点 T。源点 S 有无限多的水流可以向外流出，汇点 T 可以接受无限多的水流。其中对于每一条有向边有一个边权代表这条有向边最大可以流过的流量(用 c(u, v) 来表示)。

1.2可行流，不考虑反向边

对于可行流f来表示，f指的是对于一整张有向图而言，指定有向图中每条有向边流过多少流量。f(u,v) 指的是从 u 到 v 的一条有向边的可行流，c(u,v)指的是容量

1.2.1 f满足两个条件：容量限制，流量

1、容量限制： 0 <= f(u,v) <= c(u,v)

2、流量守恒：$\forall_x \in(V - \lbrace S,T \rbrace) \sum\forall(v,x)\in E f(v,x) = \sum\forall(x,v)\in E f(x,v) $

1.2.2 可行流的流量是指从源点流出的流量 - 流入源点的流量

$|f| = \sum_{(s,v)\in E} f(s,v) - \sum_{(v,s)\in E} f(v,s)$

1.2.3 最大流是指最大可行流

1.3 残留网络，考虑反向边

残留网络指的是在原网络中指定了可行流之后对应的一个网络流，原网络的一个可行流对应一个残留网络。其中残留网络中的点包含原网络中所有点V、边包含了原网络中的边E和其一一对应的反向边RE

1.3.1 残留网络中的可行流 f’ + 原图中的可行流 f = 原题中的一个可行流

其中残留网络的可行流 f’ 需要满足：

$ 1、容量限制：0 \leq f’(u,v) \leq c’(u,v) $

$2、流量守恒：\forall_x \in (V - \lbrace S,T \rbrace) \sum \forall(v,x) \in E f’(v,x) = \sum \forall(v,x) \in E f’(x,v)$

f’ + f 需要满足

结论 残留网络中的任何一个f’ ， |f’| 大于0， 都可以更新源网络的可行流。

1.4增广路径

增广路径指的是在残留网络中一条路经从源点到汇点，其中的每一条边的参与容量都大于等于0.

1.5割

1.5.1 割的定义

将原网络中的V 分成两个子集S和T。且满足割的选择有很多种

$ S \bigcup T = V, S \bigcap T = \emptyset $

$ 同时源点 \in S , 汇点 \in T $

1.5.2 割的容量，不考虑反向边，“最小割”是指容量最小的割

$ 割的容量c = \sum_{\forall_u\in S} \sum_{\forall_v \in T} c(u,v)[(u,v) \in E] $

1.5.3 割的流量，考虑反向边， f(S,T) <= c(S,T) ​ , f(S,T) = |f| => |f| <= c(S,T) ​

$割的流量f = \sum_{\forall_u\in S} \sum_{\forall_v \in T} c(u,v)[(u,v) \in E] - 割的容量c = \sum_{\forall_v\in S} \sum_{\forall_u \in T} c(v,u)[(v,u) \in E]$

性质：

割流量 = 可行流

1.5.6 最大流最小割定理

对于一个网络流而言

(1)可行流f是最大流

(2)可行流f的存留网络中不存在增广路径

(3) 存在某个割[S,T], |f| = c(S,T)

最大流 = 最小割

2.具体算法

2.1 EK算法 —— 维护的是残留网络

int EK()
{
    int res = 0;
    while(bfs()){
        res += d[T]；
        for(int i = T; i != S ;i = e[pre[i] ^ 1]){
            f[pre[i]] -= d[T], f[pre[i] ^ 1] += d[T];
        }
    }
}

// bfs 寻找增广路径， res 记录最大流 for循环更新 残留网络
#include <bits/stdc++.h>

using namespace std;

const int N = 1010, M = 100010, INF = 1e8;

int e[M],ne[M],h[N],f[M],idx;
int pre[N];
int d[N];
bool st[N];
int q[N];
int n,m,S,T;

void add(int a,int b,int c)
{
    e[idx] = b, f[idx] = c, ne[idx] = h[a] ,h[a] = idx ++;
}
bool bfs() // 寻找增广路径
{
    int hh = 0, tt = 0;
    memset(st,0,sizeof st);
    q[0] = S, st[S] = true, d[S] = INF;

    while(hh <= tt){
        int t = q[ hh ++];

        for(int i = h[t]; ~i; i = ne[i]){
            int j = e[i];
            if(!st[j] && f[i]){
                st[j] = true;
                pre[j] = i;
                d[j] = min(d[t],f[i]); //寻找最小的可流量

                if(j == T) return true; // 存在增广路径 返回true
                q[ ++tt] = j;
            }
        }
    }
    return false;
}

int EK()
{
    int res = 0;
    while(bfs()){
        res += d[T];
        // 更新残留网络， 倒着往回走，所以是走 ^1的这条边
        for(int i = T; i != S ; i = e[pre[i] ^ 1]){
            f[pre[i]] -= d[T], f[pre[i] ^ 1] += d[T];
        }
    }
    return res;
}

int main()
{
    memset(h,-1,sizeof h);
    scanf("%d%d%d%d",&n,&m,&S,&T);
    for(int i = 1; i <= m; i ++){
        int a,b,c; scanf("%d%d%d",&a,&b,&c);
        add(a,b,c);
        add(b,a,0);
    }

    printf("%d\n",EK());
    return 0;
}

#include <bits/stdc++.h>

using namespace std;
const int N = 10010, M = 200010, INF = 1e8;
int e[M],ne[M],h[N],f[M],idx;
int d[N],cur[N],q[N];
int n,m,S,T;

void add(int a,int b,int c)
{
    e[idx] = b, ne[idx] = h[a], f[idx] = c, h[a] = idx ++;
}

bool bfs()
{
    int hh = 0, tt = 0;
    memset(d,-1,sizeof d);
    q[0] = S, cur[S] = h[S], d[S] = 0;

    while(hh <= tt){
        int t = q[ hh ++];

        for(int i = h[t] ; ~i ; i = ne[i]){
            int j = e[i];
            if(d[j] == -1 && f[i]){
                d[j] = d[t] + 1;
                cur[j] = h[j];

                if(j == T) return true;
                q[ ++tt] = j;
            }
        }
    }
    return false;
}

int find(int u,int limit)
{
    if(u == T) return limit;

    int flow = 0;
    for(int i = cur[u] ; ~i && flow < limit ;i = ne[i]){
        int j = e[i];
        cur[u] = i;
        if(d[j] == d[u] + 1 && f[i]){
            int t = find(j,min(f[i] ,limit - flow));
            if(!t) d[j] = -1;
            f[i] -= t, f[i ^ 1] += t, flow += t;
        }
    }
    return flow;
}

int dinic()
{
    int res = 0;
    while(bfs()) while(int flow = find(S,INF)) res += flow;
    return res;
}

int main()
{
    scanf("%d%d%d%d",&n,&m,&S,&T);
    memset(h,-1,sizeof h);

    for(int i = 1; i <= m; i ++){
        int a,b,c; scanf("%d%d%d",&a,&b,&c);
        add(a,b,c);
        add(b,a,0);
    }

    printf("%d\n",dinic());
    return 0;
}
